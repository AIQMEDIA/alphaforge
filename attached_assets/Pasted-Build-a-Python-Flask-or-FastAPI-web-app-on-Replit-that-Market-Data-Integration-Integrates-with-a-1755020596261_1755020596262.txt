Sample Starter Code Block:

python
# main.py (Python FastAPI/Flask)
from fastapi import FastAPI
import requests
import os

ALPHA_VANTAGE_API_KEY = os.environ.get("ALPHA_VANTAGE_API_KEY")
FINNHUB_API_KEY = os.environ.get("FINNHUB_API_KEY")

app = FastAPI()

@app.get("/alpha_vantage")
def alpha_vantage_quote(symbol: str):
    url = f"https://www.alphavantage.co/query?function=TIME_SERIES_INTRADAY&symbol={symbol}&interval=1min&apikey={ALPHA_VANTAGE_API_KEY}"
    response = requests.get(url)
    return response.json()

@app.get("/finnhub")
def finnhub_quote(symbol: str):
    url = f"https://finnhub.io/api/v1/quote?symbol={symbol}&token={FINNHUB_API_KEY}"
    response = requests.get(url)
    return response.json()
