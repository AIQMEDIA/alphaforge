activity_logger.py
python
import json
from datetime import datetime

LOG_FILE = "trade_activity_log.json"

def log_activity(trader_type, details):
    entry = {
        "timestamp": datetime.utcnow().isoformat(),
        "trader_type": trader_type,
        "details": details
    }
    try:
        with open(LOG_FILE, "a") as f:
            f.write(json.dumps(entry) + "\n")
    except Exception as e:
        print(f"Logging error: {e}")
trade_analyzer.py
python
def classify_trader(activity):
    # Heuristics for classifying trading behavior
    if activity["order_size"] >= 10000 or activity.get("algo_type") in ["VWAP", "TWAP", "Iceberg"]:
        return "institutional"
    if activity["venue"] in ["dark_pool", "direct_market_access"]:
        return "institutional"
    if activity["frequency_per_hour"] > 50:
        return "institutional"
    # Retail indicators
    if activity["order_size"] <= 5000 and activity.get("algo_type") is None:
        return "retail"
    if activity["app_type"] == "mobile" or activity["venue"] == "retail_broker":
        return "retail"
    return "uncertain"

def analyze_trade(activity):
    # Optionally, add further analytics: e.g., session metadata, timing, etc.
    analysis = {
        "order_size": activity["order_size"],
        "algo_type": activity.get("algo_type"),
        "venue": activity.get("venue"),
        "app_type": activity.get("app_type"),
        "frequency_per_hour": activity.get("frequency_per_hour", 0),
    }
    trader_type = classify_trader(activity)
    return trader_type, analysis

How This Works
Feed trade/activity data into main.py (live or batch).

The system analyzes each activity according to simple heuristics (size, algo, venue, app type, frequency).

Logs all activity to a JSON file for further review, auditing, or ML training.

Easily extensible to include:

Session data (IP, latency, registration).

Call to notification system for suspicious or institutional traffic.

API hooks for feeding into dashboards or anomaly detection systems.
