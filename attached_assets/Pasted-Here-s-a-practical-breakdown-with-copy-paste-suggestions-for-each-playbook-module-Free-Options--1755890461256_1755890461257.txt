Example: Simple IP Rate Limit

python
from collections import defaultdict
import time
ip_hits = defaultdict(list)
def rate_limit(ip):
    now = time.time()
    ip_hits[ip] = [t for t in ip_hits[ip] if now - t < 60]
    ip_hits[ip].append(now)
    return len(ip_hits[ip]) > 10  # block if >10 requests/min

logger.py
python
import sqlite3
import datetime
import json

DB = 'activity_log.db'

def init_db():
    conn = sqlite3.connect(DB)
    c = conn.cursor()
    c.execute('CREATE TABLE IF NOT EXISTS events (timestamp TEXT, event_type TEXT, details TEXT)')
    conn.commit()
    conn.close()

def log_event(event_type, details):
    conn = sqlite3.connect(DB)
    c = conn.cursor()
    c.execute(
        'INSERT INTO events VALUES (?, ?, ?)',
        (datetime.datetime.utcnow().isoformat(), event_type, json.dumps(details))
    )
    conn.commit()
    conn.close()
    print(f"[LOGGED] [{event_type}] {details}")
notifier.py
python
import requests

# --- Slack Alert (free, with webhooks) ---
SLACK_HOOK = "https://hooks.slack.com/services/PUT/REAL/WEBHOOK"  # replace with your Slack Webhook URL

def slack_alert(msg):
    try:
        requests.post(SLACK_HOOK, json={"text": msg})
        print("[ALERT] Sent to Slack")
    except Exception as e:
        print(f"[ALERT ERROR] {e}")

# --- Email: For Replit, Mailgun Free Tier or Gmail SMTP (see docs for security advice) ---
import smtplib
from email.mime.text import MIMEText

def email_alert(subject, body, to_email):
    # For demo only; use app password and secure storage!
    SMTP_SERVER = "smtp.gmail.com"
    SMTP_PORT = 587
    GMAIL_USER = "your_email@gmail.com"
    GMAIL_PASS = "your_app_password"
    msg = MIMEText(body)
    msg['Subject'] = subject
    msg['From'] = GMAIL_USER
    msg['To'] = to_email
    try:
        server = smtplib.SMTP(SMTP_SERVER, SMTP_PORT)
        server.starttls()
        server.login(GMAIL_USER, GMAIL_PASS)
        server.sendmail(GMAIL_USER, [to_email], msg.as_string())
        server.quit()
        print("[ALERT] Email sent")
    except Exception as e:
        print(f"[ALERT ERROR] {e}")

session_manager.py
python
from collections import defaultdict
import time

SESSION_CONTROL = {}

def restrict_session(session_id, label, reason):
    SESSION_CONTROL[session_id] = {
        "status": "RESTRICTED",
        "reason": reason,
        "label": label,
        "restricted_at": time.time()
    }
    print(f"[SESSION] {session_id} restricted: {reason}")

ip_hits = defaultdict(list)
def rate_limit(ip):
    now = time.time()
    ip_hits[ip] = [t for t in ip_hits[ip] if now-t < 60]
    ip_hits[ip].append(now)
    return len(ip_hits[ip]) > 10  # e.g., block if more than 10 req/min
main.py
python
from logger import init_db, log_event
from notifier import slack_alert, email_alert
from bi_router import log_prospect_to_csv
from session_manager import restrict_session, rate_limit

# --- Init logging db ---
init_db()

# --- Example suspicious event ---
event = {
    "session_id": "abc123",
    "risk_score": 180,
    "trader_type": "hedge_fund",
    "business_value_score": 160,
    "full_trace": "...",
    "algo": "TWAP",
    "app_type": "desktop",
    "ip": "204.68.37.1"
}

# --- 1: Restrict session if needed ---
if event["risk_score"] > 150:
    restrict_session(event["session_id"], event["trader_type"], "High-risk institutional algo detected")
    log_event("session_restricted", event)

# --- 2: Log event for analysis ---
log_event("suspicious_trade_bot", event)

# --- 3: Send free alert (Slack) ---
slack_alert(f"🚨 Detected high-risk {event['trader_type']} bot! Session: {event['session_id']} Risk: {event['risk_score']}")

# --- 4: (Optionally) Email admin (uncomment if needed) ---
# email_alert(
#     subject="AlphaForge ALERT: High-Risk Trading Bot",
#     body=f"Alert: suspicious {event['trader_type']} bot detected.\nDetails: {event}",
#     to_email="admin@yourdomain.com"
# )

# --- 5: BI Prospect Routing (CSV) ---
if event["business_value_score"] > 100:
    log_prospect_to_csv([
        event["session_id"],
        event["trader_type"],
        event["business_value_score"],
        event.get("algo", ""),
        event.get("app_type", ""),
        event.get("ip", "")
    ])

# --- 6: Free IP rate limiting example ---
if rate_limit(event["ip"]):
    print(f"[RATE LIMIT] Blocking/further restricting IP: {event['ip']}")

print("\nWorkflow completed! All actions logged and free alerts sent.")
