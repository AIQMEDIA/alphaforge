test_api_core.py
python
import pytest
import requests

API_BASE = "http://localhost:8000/api"
FAKE_TOKEN = "demo_token"  # Use your Replit Auth token here

def get_headers():
    return {"Authorization": f"Bearer {FAKE_TOKEN}"}

def test_api_status():
    r = requests.get(f"{API_BASE}/status", headers=get_headers())
    assert r.status_code == 200
    assert "uptime" in r.json()

def test_markets_listing():
    r = requests.get(f"{API_BASE}/markets", headers=get_headers())
    assert r.status_code == 200
    assert isinstance(r.json(), list)

def test_portfolio_crud():
    # Create
    data = {"name": "Test Portfolio", "assets": ["AAPL", "MSFT"]}
    r = requests.post(f"{API_BASE}/portfolio", json=data, headers=get_headers())
    assert r.status_code == 201
    pid = r.json().get("id")

    # Retrieve
    r = requests.get(f"{API_BASE}/portfolio/{pid}", headers=get_headers())
    assert r.status_code == 200
    assert r.json()["name"] == "Test Portfolio"

    # Delete
    r = requests.delete(f"{API_BASE}/portfolio/{pid}", headers=get_headers())
    assert r.status_code == 204

import pytest

from faker import Faker

API_BASE = "http://localhost:8000/api"

    # Post strategy as user #1
    strat = {
        "name": fake.catch_phrase(),
        "script": "quantum_strategy_vqe()",
        "description": fake.text(),
    }
    headers = {"Authorization": "Bearer user1_token"}
    r = requests.post(f"{API_BASE}/marketplace/strategies", json=strat, headers=headers)
    assert r.status_code == 201
    sid = r.json()["id"]

    # Clone as user #2
    headers2 = {"Authorization": "Bearer user2_token"}

test_security_fraud.py
python
import pytest
import requests

API_BASE = "http://localhost:8000/api"

def test_fraud_trial_eligibility():
    # Simulated multiple signups from same device/IP
    device_id = "FAKE_FINGERPRINT"
    headers = {"X-Device-ID": device_id}
    for _ in range(5):
        r = requests.post(f"{API_BASE}/auth/signup", json={"email": f"fraud{_}@disposable.com"}, headers=headers)
    # The 4th-5th attempt should be blocked or throttled
    r = requests.post(f"{API_BASE}/auth/signup", json={"email": "fraud999@disposable.com"}, headers=headers)
    assert r.status_code in (403, 429)

def test_rate_limit_bots():
    headers = {"Authorization": "Bearer valid_user_token"}
    for _ in range(12):
        r = requests.post(f"{API_BASE}/marketdata/price", json={"symbol": "AAPL"}, headers=headers)
    # Should block further attempts after threshold
    r = requests.post(f"{API_BASE}/marketdata/price", json={"symbol": "TSLA"}, headers=headers)
    assert r.status_code in (429, 403)

def test_disposable_email_block():
    headers = {"Authorization": "Bearer valid_user_token"}
    r = requests.post(f"{API_BASE}/profile/update_email", json={"email": "test@mailinator.com"}, headers=headers)
    assert r.status_code in (400, 403)
test_end2end_playwright.py
This file uses Playwright for real browser-based E2E flow.
First, run:

shell
playwright install
Then use:

python
import pytest
from playwright.sync_api import sync_playwright

def test_signup_and_quantum_trade_flow():
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()
        page.goto("http://localhost:8000")
        # Simulate signup
        page.click("text=Sign Up")
        page.fill("input[name='email']", "e2euser@example.com")
        page.fill("input[name='password']", "StrongPassword123")
        page.click("text=Register")
        # Ensure portfolio creation
        page.click("text=Create Portfolio")
        page.fill("input[name='name']", "E2E Quantum")
        page.click("text=Save")
        # Run quantum optimization
        page.click("text=Quantum Optimize")
        page.wait_for_selector("text=Optimization Complete", timeout=15000)
        # Confirm results shown
        assert page.is_visible("text=Expected Return")
        browser.close()
